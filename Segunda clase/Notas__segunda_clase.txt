fun bucleFor(){
    val notas = intArrayOf(13, 20, 4, 6, 8)
    for(nota: Int in notas){
        println/(nota.toString())
    } // toma dato por valor e indice
    for((indice, valor) in notas.withIndex()){
        println("Nota $indice es igual $valor")
    } // se puede recorrer intervalos de nros
    for(numero in 5 .. 15){
        println("Numero $numero")
    }
}
